 # 测试策略详细设计文档

基于代码架构和潜在风险点，以下是完整的测试方案，按模块分层设计。

---

## 1. 测试框架选型

```go
// 推荐依赖（添加到 go.mod）
require (
    github.com/stretchr/testify v1.9.0  // 断言库
    github.com/xuri/excelize/v2 v2.9.0  // 已存在，用于生成测试Excel
)
```

**选型理由：**
- `testify`：提供 `assert` 和 `require`，支持表格驱动测试
- 不引入 mock 框架（如 gomock），因为本工具主要是数据转换，依赖真实 Excel 文件更可靠

// schema/parser_test.go
func TestParseHeader(t *testing.T) {
    tests := []struct {
        name        string
        rows        [][]string
        wantErr     bool
        errContains string
        validate    func(t *testing.T, s *SheetSchema)
    }{
        {
            name: "标准4行表头",
            rows: [][]string{
                {"ItemConfig", "道具ID", "道具名称"},  // Label行
                {"", "int", "string"},                   // Type行
                {"", "Client", "Client"},                // Client行（被忽略）
                {"", "Server", "Server"},                // FieldName行标识
                {"", "id", "name"},                      // FieldName行
            },
            wantErr: false,
            validate: func(t *testing.T, s *SheetSchema) {
                assert.Equal(t, "ItemConfig", s.ClassName)
                assert.Equal(t, 2, len(s.Fields))
                assert.Equal(t, "id", s.Fields[0].FieldName)
                assert.Equal(t, TypeInt, s.Fields[0].FieldType)
                assert.Equal(t, 1, s.Fields[0].ColIndex)  // B列
            },
        },
        {
            name: "FieldName行不在第4行（动态定位）",
            rows: [][]string{
                {"ItemConfig", "备注"},      // 额外行
                {"", "说明"},                // 额外行
                {"", "int"},                 // Type行
                {"", "Server"},              // 标识行（第4行）
                {"", "id"},                  // FieldName行（第5行）
            },
            wantErr: false,
            validate: func(t *testing.T, s *SheetSchema) {
                assert.Equal(t, 5, s.DataStartRow)  // 数据从第6行开始
            },
        },
        {
            name: "A1为空",
            rows: [][]string{
                {"", "道具ID"},  // A1为空
                {"", "int"},
                {"", "Server"},
                {"", "id"},
            },
            wantErr:     true,
            errContains: "A1 为空",
        },
        {
            name: "缺少Server标识行",
            rows: [][]string{
                {"ItemConfig", "道具ID"},
                {"", "int"},
                {"", "id"},  // 直接写字段名，没有Server标识
            },
            wantErr:     true,
            errContains: "未找到 FieldName 行",
        },
        {
            name: "嵌套字段识别",
            rows: [][]string{
                {"ItemConfig", "奖励"},
                {"", "[]struct"},
                {"", "Server"},
                {"", "rewards[0].id"},
            },
            wantErr: false,
            validate: func(t *testing.T, s *SheetSchema) {
                assert.True(t, isNestedField(s.Fields[0].FieldName))
            },
        },
        {
            name: "ignore类型列",
            rows: [][]string{
                {"ItemConfig", "道具ID", "备注"},
                {"", "int", "ignore"},
                {"", "Server", "Server"},
                {"", "id", "#remark"},
            },
            wantErr: false,
            validate: func(t *testing.T, s *SheetSchema) {
                assert.Equal(t, 1, len(s.Fields))  // 只保留id列
                assert.True(t, s.Fields[0].FieldName != "#remark")
            },
        },
        {
            name: "空Sheet（无数据行）",
            rows: [][]string{
                {"ItemConfig", "道具ID"},
                {"", "int"},
                {"", "Server"},
                {"", "id"},
                // 无数据行
            },
            wantErr: false,
            validate: func(t *testing.T, s *SheetSchema) {
                assert.Equal(t, 0, len(s.DataRows))
            },
        },
        {
            name: "列数不一致（稀疏表）",
            rows: [][]string{
                {"ItemConfig", "A", "B", "C", "D", "E"},  // 5列
                {"", "int"},                               // 只有1列类型
                {"", "Server"},
                {"", "id"},                                // 只有1列字段名
            },
            wantErr: false,
            validate: func(t *testing.T, s *SheetSchema) {
                assert.Equal(t, 1, len(s.Fields))  // 只解析到有定义的列
            },
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            schema, err := ParseHeader(tt.rows, "test.xlsx", "Sheet1")
            if tt.wantErr {
                assert.Error(t, err)
                if tt.errContains != "" {
                    assert.Contains(t, err.Error(), tt.errContains)
                }
                return
            }
            assert.NoError(t, err)
            if tt.validate != nil {
                tt.validate(t, schema)
            }
        })
    }
}

func TestParsePath(t *testing.T) {
    tests := []struct {

            fieldName: "rewards[0]",
            wantSegments: []PathSegment{
                {Name: "rewards", ArrayIdx: 0, MapKey: -1, IsArray: true, IsMap: false},
            },
        },
        {
            name:      "数组元素属性",
            fieldName: "rewards[0].id",
            wantSegments: []PathSegment{
                {Name: "rewards", ArrayIdx: 0, MapKey: -1, IsArray: true, IsMap: false},
                {Name: "id", ArrayIdx: -1, MapKey: -1, IsArray: false, IsMap: false},
            },
        },
        {
            name:      "多层数组",
            fieldName: "matrix[0][1]",  // 注意：实际语法是 matrix[0].[1] 还是 matrix[0][1]？
            fieldName: "matrix[0].inner[1]",
            wantSegments: []PathSegment{
                {Name: "matrix", ArrayIdx: 0, IsArray: true},
                {Name: "inner", ArrayIdx: 1, IsArray: true},
            },
        },
        {
            name:      "Map语法",
            fieldName: "bonus{100}.value",
            wantSegments: []PathSegment{
                {Name: "bonus", ArrayIdx: -1, MapKey: 100, IsArray: false, IsMap: true},
                {Name: "value", ArrayIdx: -1, MapKey: -1, IsArray: false, IsMap: false},
            },
        },
        {
            name:      "混合嵌套",
            fieldName: "items[0].attrs{1}.value",
            wantSegments: []PathSegment{
                {Name: "items", ArrayIdx: 0, IsArray: true},
                {Name: "attrs", MapKey: 1, IsMap: true},
                {Name: "value"},
            },
        },
        {
            name:      "超大数组索引",
            fieldName: "arr[999999]",
            wantSegments: []PathSegment{
                {Name: "arr", ArrayIdx: 999999, IsArray: true},
            },
        },
        {
            name:      "负数索引（边界）",
            fieldName: "arr[-1]",
            wantErr:   true,  // 应该报错
        },
        {
            name:      "空数组索引",
            fieldName: "arr[]",
            wantErr:   true,
        },
        {
            name:      "非数字索引",
            fieldName: "arr[abc]",
            wantErr:   true,
        },
        {
            name:      "连续点号",
            fieldName: "a..b",
            wantSegments: []PathSegment{  // 实际行为取决于实现
                {Name: "a"},
                {Name: ""},  // 空字符串段
                {Name: "b"},
            },
        },
        {
            name:      "空字符串",
            fieldName: "",
            wantErr:   true,
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            segments, err := ParsePath(tt.fieldName)
            if tt.wantErr {
                assert.Error(t, err)
                return
            }
            assert.NoError(t, err)
            assert.Equal(t, tt.wantSegments, segments)
        })
    }

func TestSetValueByPath(t *testing.T) {
    tests := []struct {
        name     string
        segments []PathSegment
        value    interface{}
        initial  map[string]interface{}  // 初始状态
        expected map[string]interface{}
        wantErr  bool
    }{
        {
            name:     "设置简单字段",
            segments: []PathSegment{{Name: "id"}},
            value:    1001,
            initial:  map[string]interface{}{},
            expected: map[string]interface{}{"id": 1001},
        },
        {
            name: "创建数组并设置元素",
            segments: []PathSegment{
                {Name: "rewards", ArrayIdx: 0, IsArray: true},
                {Name: "id"},
            },
            value:   2001,
            initial: map[string]interface{}{},
            expected: map[string]interface{}{
                "rewards": []interface{}{
                    map[string]interface{}{"id": 2001},
                },
            },
        },
        {
            name: "数组自动扩容",
            segments: []PathSegment{
                {Name: "items", ArrayIdx: 5, IsArray: true},  // 索引5，前面补nil
            },
            value:   "value",
            initial: map[string]interface{}{},
            expected: map[string]interface{}{
                "items": []interface{}{nil, nil, nil, nil, nil, "value"},
            },
        },
        {
            name: "深层嵌套创建",
            segments: []PathSegment{
                {Name: "a", ArrayIdx: 0, IsArray: true},
                {Name: "b", MapKey: 1, IsMap: true},
                {Name: "c"},
            },
            value:   "deep",
            initial: map[string]interface{}{},
            expected: map[string]interface{}{
                "a": []interface{}{
                    map[string]interface{}{
                        "b": map[int]interface{}{
                            1: map[string]interface{}{"c": "deep"},
                        },
                    },
                },
            },
        },
        {
            name: "路径冲突-重复赋值相同值",
            segments: []PathSegment{{Name: "id"}},
            value:    1001,
            initial:  map[string]interface{}{"id": 1001},
            expected: map[string]interface{}{"id": 1001},  // 应该允许
            wantErr:  false,
        },
        {
            name: "路径冲突-重复赋值不同值",
            segments: []PathSegment{{Name: "id"}},
            value:    1002,
            initial:  map[string]interface{}{"id": 1001},
            wantErr:  true,  // 应该报错
        },
        {
            name: "类型冲突-对象变数组",
            segments: []PathSegment{
                {Name: "data", ArrayIdx: 0, IsArray: true},
            },
            value:   1,
            initial: map[string]interface{}{"data": map[string]interface{}{}},
            wantErr: true,
        },
        {
            name: "Map键设置",
            segments: []PathSegment{
                {Name: "config", MapKey: 100, IsMap: true},
            },
            value:   "config100",
            initial: map[string]interface{}{},
            expected: map[string]interface{}{
                "config": map[int]interface{}{100: "config100"},
            },
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            data := tt.initial
            if data == nil {
                data = make(map[string]interface{})
            }
            err := SetValueByPath(data, tt.segments, tt.value)
            if tt.wantErr {
                assert.Error(t, err)
                return
            }
            assert.NoError(t, err)
            assert.Equal(t, tt.expected, data)
        })
    }
}

## 5. 优先级建议

| 优先级 | 模块 | 理由 |
|-------|------|------|
| P0 | `builder/path.go` | 核心复杂逻辑，嵌套路径易出错 |
| P0 | `schema/parser.go` | 表头解析是所有后续步骤的基础 |
| P1 | `validator/validator.go` | 数据质量保证，主键重复检测关键 |
| P1 | `builder/builder.go` | 类型转换逻辑复杂 |
| P2 | `classconfig/parser.go` | 配置解析相对独立 |
| P2 | `merger/merger.go` | 逻辑较简单，但排序规则需验证 |
| P3 | `exporter/json.go` | 主要是标准库调用 |
| P3 | 集成测试 | 确保全流程通畅 |

这套测试方案覆盖了单元测试、边界测试和集成测试，特别关注了数组索引溢出、深层嵌套、类型转换等边界情况。建议先实现 P0/P1 级别的测试，确保核心逻辑稳固。
